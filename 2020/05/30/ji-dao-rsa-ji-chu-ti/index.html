def CRT(bs,ms):
    m = reduce(lambda x,y: x*y, ms)
    re = 0
    for i in range(len(ms)):
        M = m / ms[i]
        gcd, n1, M1 = gmpy2.gcdext(ms[i], M)# ms[i]*n1 + M*M1 = 1
        re += bs[i] * M * M1
    return re % m
